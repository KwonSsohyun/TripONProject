SELECT * FROM PURCHASE;
DELETE FROM PURCHASE;
DROP TABLE PURCHASE;

CREATE TABLE PURCHASE (
PUR_NUM INT AUTO_INCREMENT PRIMARY KEY,
PUR_STATENUM INT NOT NULL,
PUR_IMPID VARCHAR(30) UNIQUE NOT NULL,
PUR_MERCHANTID VARCHAR(30) NOT NULL,
PUR_USER_ID VARCHAR(20) NOT NULL,
PUR_USER_NAME VARCHAR(20) NOT NULL,
PUR_PURCHASEDATE VARCHAR(30) NOT NULL,
PUR_BOOKINGDATE VARCHAR(30) NOT NULL,
PUR_ACT_TITLE VARCHAR(200) NOT NULL,
PUR_OPT_NAME_1 VARCHAR(50),
PUR_OPT_NAME_2 VARCHAR(50),
PUR_OPT_NAME_3 VARCHAR(50),
PUR_OPT_NAME_4 VARCHAR(50),
PUR_OPT_COUNT_1 INT,
PUR_OPT_COUNT_2 INT,
PUR_OPT_COUNT_3 INT,
PUR_OPT_COUNT_4 INT,
PUR_PRICE INT NOT NULL
)DEFAULT CHARSET=UTF8;
